def cbs(cols):
    newRow = pd.DataFrame(columns=cols.columns)
    for index, row in cols.iterrows():
        tmp = row[1].split(";")
        for item in tmp:
            newRow.loc[len(newRow.index)] = [row[0], item]
    return(newRow)
